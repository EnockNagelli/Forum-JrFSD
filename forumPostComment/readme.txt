   1. PostController	 				- PORT: 8092
//----------------------------------------------------------------------------------------------------------------------------------
	 SERVICE OPERATIONS
//----------------------------------------------------------------------------------------------------------------------------------
	@RequestMapping (value = "/") 		- return "Welcome to Forum Application - Post Service : 
											      You can Post and/or Comment and/or Like your opinion";
	//--------------------------------------------------------------------------------------------------------------------------
	@RequestMapping (value = "/post") 	- return "Welcome to Forum Application - Post Service : 
												  You can 'Post' your opinion";
	//--------------------------------------------------------------------------------------------------------------------------
	@PostMapping(value="/post/newPost/{userId}/{categoryId}")				- ResponseEntity<Boolean> 
	//--------------------------------------------------------------------------------------------------------------------------
	@DeleteMapping(value = "/post/deletePost/{postId}") 					- ResponseEntity<Boolean> 
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/post/getPostById/{postId}") 						- ResponseEntity<VisitorPostDto> 
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/post/getPostByCategory/{category}") 				- ResponseEntity<List<VisitorPostDto>>
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/post/getAllPosts", produces = "application/json") - ResponseEntity<List<VisitorPostDto>> 
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/post/getDiscussionList") 							- ResponseEntity<Map<String, String>>	
//----------------------------------------------------------------------------------------------------------------------------------
	// SEARCH OPERATIONS
//----------------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/search/post/{postId}") 				- ResponseEntity<VisitorPostDto> 	
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/search/category/{category}") 			- ResponseEntity<List<VisitorPostDto>>
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/search/title/{title}") 				- ResponseEntity<List<VisitorPostDto>> 
	//--------------------------------------------------------------------------------------------------------------------------
	@GetMapping(value = "/search/tag/{tag}") 					- ResponseEntity<List<VisitorPostDto>>
